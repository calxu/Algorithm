LCSS(X,Y)
  m = X.length
  n = Y.length
  Let b[1..m, 1..n] and c[0..m, 0..n] be new tables
  for i = 1 to m
c[i, 0] = 0
  for j = 0 to n
c[0,j] = 0
  for i = 1 to m
for j = 1 to n
  If x[i] == y[j]
  c[i, j] = c[i-1, j-1] + 1
  elseif c[i-1, j] >= c[i, j-1]
  c[i, j] = c[i-1, j]
  else c[i, j] = c[i, j-1]
